api/lib/storage.js

/**
 * window.localStorage 的 get 和 set
 * @memberof $api
 * @module storage
 * @summary 简单本地存储
 */

var storage = {}
/**
 * 通过键(key)检索获取应用存储的值
 * @public
 * @function get
 * @param {string} key
 * @returns {string}
 * @example
 * $api.storage.set('a',JSON.stringify({a:1}))
 * JSON.parse($api.storage.get('a'))
 */
storage.get = function(key) {
  return window.localStorage.getItem(key) || ''
}
/**
 * 修改或添加键值(key-value)对数据到应用数据存储中
 * @public
 * @function set
 * @param {string} key
 * @param {string} value
 * @example
 * $api.storage.set('a',JSON.stringify({a:1}))
 */
storage.set = function set(key, value) {
  window.localStorage.setItem(key, value)
}
/**
 * 通过键(key)删除
 * @public
 * @function del
 * @param {string} key
 * @example
 * $api.storage.set('a',JSON.stringify({a:1}))
 * JSON.parse($api.storage.get('a'))
 * $api.storage.del('a')
 */
storage.del = function(key) {
  return window.localStorage.removeItem(key)
}

if (!window.$api.storage) {
  Object.defineProperty(window.$api, 'storage', {
    configurable: false,
    enumerable: false,
    Writable: false,
    get: function() {
      return storage
    }
  })
}
