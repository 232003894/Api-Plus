/**
 * 图片操作读取base64
 * @module img
 * @memberof $api
 * @summary 图片操作读取base64
 */

var img = {}

/**
 * 通过路径加载图片，获取base64格式的图片数据
 * @public
 * @function load
 * @param {string} src 图片路径，本地或远程图片
 * @param {number} [width] 图片宽度，如果不传将使用图片的原始宽度
 * @param {number} [height] 图片高度，如果不传将使用图片的原始高度
 * @return {Promise}
 * @example
 * $api.img.load(url).then(base64 => {
 *  console.log(base64.length / 1024)
 * })
 */
img.load = function(src, width, height) {
  return new Promise(function(resolve, reject) {
    var _img = new Image()
    _img.setAttribute('crossOrigin', 'Anonymous')
    // console.log(src)
    _img.src = src
    _img.onload = function() {
      return resolve(img.base64(_img, width, height))
    }
    _img.onerror = reject
  })
}

/**
 * 通过路径加载图片，获取base64格式的图片数据
 * @public
 * @function base64
 * @param {img} img 图片对象
 * @param {number} [width] 图片宽度，如果不传将使用图片的原始宽度
 * @param {number} [height] 图片高度，如果不传将使用图片的原始高度
 * @returns {string} base64字符串
 * @example
 * $api.img.base64(img)
 */
img.base64 = function(_img, width, height) {
  var canvas = document.createElement('canvas')
  canvas.width = width ? width : _img.width
  canvas.height = height ? height : _img.height
  var ctx = canvas.getContext('2d')
  ctx.drawImage(_img, 0, 0, canvas.width, canvas.height)
  var dataURL = canvas.toDataURL()
  // console.log('压缩后：' + dataURL.length / 1024)
  return dataURL
}

if (!window.$api.img) {
  Object.defineProperty(window.$api, 'img', {
    configurable: false,
    enumerable: false,
    Writable: false,
    get: function() {
      return img
    }
  })
}
