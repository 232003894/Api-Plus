/**
 * Dom加载完成，相当于window.onload
 * @memberof $api
 * @name ready
 * @param {Function} callback - 回调
 * @param {Function} [vm] - 上下文对象，this指向的对象
 * @example <caption>Dom加载完成</caption>
 * $api.ready(function(){
 *   console.log('Dom 加载完成')
 * })
 */
function onload(callback, vm) {
  if (vm === undefined) {
    vm = null
  }
  var readyRE = /complete|loaded|interactive/
  if (readyRE.test(document.readyState)) {
    if (vm) {
      callback.call(vm)
    } else {
      callback()
    }
  } else {
    document.addEventListener(
      'DOMContentLoaded',
      function() {
        if (vm) {
          callback.call(vm)
        } else {
          callback()
        }
      },
      false
    )
  }
  return this
}

if (!window.$api.ready) {
  Object.defineProperty(window.$api, 'ready', {
    configurable: false,
    enumerable: false,
    Writable: false,
    get: function() {
      return onload
    }
  })
}
